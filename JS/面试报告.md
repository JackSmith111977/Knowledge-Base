#### 1. JavaScript 数据类型完整清单

**问题**：JavaScript 有哪些数据类型？

**完整答案**：

JavaScript 有 **8 种** 数据类型，分为两类：

**基本类型（Primitive Types）- 7 种**
```javascript
// 存储在栈内存中，按值访问

typeof "hello"      // "string"
typeof 42           // "number"
typeof true         // "boolean"
typeof undefined    // "undefined"
typeof null         // "object" (历史遗留 bug)
typeof Symbol()     // "symbol"
typeof 9007199254740991n  // "bigint"
```

**引用类型（Reference Types）- 1 种**
```javascript
// 存储在堆内存中，按引用访问

typeof {}           // "object"
typeof []           // "object"
typeof function(){} // "function"
typeof /regex/      // "object"
typeof new Date()   // "object"
```

**完整类型判断方案**
```javascript
function getType(value) {
  // 基本类型用 typeof（null 除外）
  if (value === null) return 'null';
  if (typeof value !== 'object') return typeof value;

  // 引用类型用 Object.prototype.toString
  return Object.prototype.toString.call(value)
    .slice(8, -1)
    .toLowerCase();
}

// 测试
getType([]);           // "array"
getType({});           // "object"
getType(function(){}); // "function"
getType(/regex/);      // "regexp"
getType(new Date());   // "date"
```

#### 2. Object.prototype.toString.call() 原理

**问题**：为什么它比 typeof 和 instanceof 更准确？实现原理是什么？

**完整答案**：

**typeof 的局限性**
```javascript
typeof null          // "object" ❌ (应该是 null)
typeof []            // "object" ❌ (无法区分数组)
typeof /regex/       // "object" ❌ (无法区分正则)
typeof new Date()    // "object" ❌ (无法区分日期)
```

**instanceof 的局限性**
```javascript
// 只能判断引用类型，基本类型全返回 false
"hello" instanceof String   // false
42 instanceof Number        // false
true instanceof Boolean     // false

// 跨 iframe/窗口时失效
iframeArray instanceof Array  // false ❌
```

**Object.prototype.toString 的原理**

每个内置类型内部都有一个 `[[Class]]` 内部槽（ES5 规范），`Object.prototype.toString` 会读取这个值：

```javascript
// 内部实现逻辑（伪代码）
Object.prototype.toString = function() {
  const classInternalSlot = this.[[Class]];
  return `[object ${classInternalSlot}]`;
};

// 各类型的 [[Class]] 值
Array.prototype.[[Class]]    // "Array"
RegExp.prototype.[[Class]]   // "RegExp"
Date.prototype.[[Class]]     // "Date"
Error.prototype.[[Class]]    // "Error"
Function.prototype.[[Class]] // "Function"
```

**为什么需要 .call()**

直接调用会获取调用对象的 [[Class]]，需要用 .call() 强制改变 this 指向：

```javascript
// 错误示范
const arr = [];
arr.toString();  // "" (调用的是 Array 的 toString)

// 正确做法
Object.prototype.toString.call(arr);  // "[object Array]"
Object.prototype.toString.call(/a/);  // "[object RegExp]"
Object.prototype.toString.call(null); // "[object Null]"
```

**完整类型判断表**
```javascript
const typeMap = {
  '[object Object]': 'object',
  '[object Array]': 'array',
  '[object Function]': 'function',
  '[object RegExp]': 'regexp',
  '[object Date]': 'date',
  '[object Error]': 'error',
  '[object String]': 'string',
  '[object Number]': 'number',
  '[object Boolean]': 'boolean',
  '[object Null]': 'null',
  '[object Undefined]': 'undefined',
  '[object Symbol]': 'symbol',
  '[object Map]': 'map',
  '[object Set]': 'set',
  '[object WeakMap]': 'weakmap',
  '[object WeakSet]': 'weakset',
  '[object Promise]': 'promise',
};

function getType(value) {
  return typeMap[Object.prototype.toString.call(value)];
}
```

#### 3. const 的"不可变"本质

**问题**：const 声明的变量真的不可变吗？

**完整答案**：

**const 保证的是绑定地址不变，不是值不变**

```javascript
// 基本类型：值就在地址里，所以看起来"不可变"
const a = 1;
a = 2;  // TypeError: Assignment to constant variable ❌

// 引用类型：地址不变，但内部属性可以修改
const obj = { a: 1 };
obj.a = 2;      // ✓ 允许！修改的是堆内存中的属性
obj.b = 3;      // ✓ 允许！添加新属性
obj = {};       // TypeError ❌ 不能改变绑定地址

// 数组同理
const arr = [1, 2];
arr.push(3);    // ✓ 允许
arr[0] = 10;    // ✓ 允许
arr = [];       // TypeError ❌
```

**如何实现真正的"不可变"**

```javascript
// Object.freeze() - 浅冻结
const obj = Object.freeze({ a: 1 });
obj.a = 2;      // 严格模式下报错，非静默失败
console.log(obj.a);  // 1

// 深度冻结
function deepFreeze(obj) {
  Object.freeze(obj);
  Object.keys(obj).forEach(key => {
    if (typeof obj[key] === 'object' && obj[key] !== null) {
      deepFreeze(obj[key]);
    }
  });
  return obj;
}

const deep = deepFreeze({ a: { b: 1 } });
deep.a.b = 2;  // 严格模式下报错
```

#### 4. 宏任务与微任务完整 API

**问题**：宏任务和微任务各有哪些？

**完整答案**：

**宏任务（Macrotask）队列**
```javascript
// 常见的宏任务 API
setTimeout(() => {}, 0);      // 延时执行
setInterval(() => {}, 1000);  // 定时执行
setImmediate(() => {});       // Node.js 专用，下次事件循环立即执行

// 宿主环境 API
// - I/O 操作（文件读写、网络请求）
// - UI 渲染（浏览器重绘、重布局）
// - user interaction（点击、键盘事件）
// - script 整体代码（首次执行）

// 示例：宏任务执行时机
console.log('start');
setTimeout(() => console.log('timeout'), 0);
console.log('end');

// 输出：start → end → timeout
```

**微任务（Microtask）队列**
```javascript
// 常见的微任务 API
Promise.resolve().then(() => {});     // Promise 回调
Promise.reject().catch(() => {});     // Promise 错误处理
Promise.all([]).then(() => {});       // Promise 集合
queueMicrotask(() => {});             // 原生微任务 API

// 特殊：Node.js
process.nextTick(() => {});           // Node 专属，优先级最高

// DOM 相关
const observer = new MutationObserver(() => {});  // DOM 变化监听
```

**执行优先级**
```
同步代码 > process.nextTick > 其他微任务 > 宏任务
```

**完整事件循环流程**
```javascript
console.log('1. script start');

setTimeout(() => {
  console.log('2. setTimeout');
}, 0);

Promise.resolve().then(() => {
  console.log('3. promise then');
});

console.log('4. script end');

// 输出顺序：
// 1. script start
// 4. script end
// 3. promise then
// 2. setTimeout
```

**为什么微任务优先？**

1. **设计目的**：微任务用于需要立即处理的异步结果
2. **状态一致性**：在下次渲染前处理完所有微任务，避免中间状态暴露
3. **链式调用**：Promise 链式调用需要连续执行

```javascript
// 示例：微任务保证状态一致性
const div = document.querySelector('#app');

Promise.resolve().then(() => {
  div.textContent = 'updated';
  // 如果微任务不优先，用户可能看到旧内容闪烁
});

// 浏览器会在渲染前清空微任务队列
// 确保用户看到的是最新状态
```

#### 5. 并发控制实现

**问题**：实现 `runLimit(tasks, limit)` 限制并发数

**完整答案**：

**方案一：递归 + Promise**

```javascript
async function runLimit(tasks, limit) {
  const results = [];
  let index = 0;

  // 执行单个任务，完成后继续下一个
  async function runNext() {
    if (index >= tasks.length) return;

    const currentIndex = index++;
    const task = tasks[currentIndex];

    // 执行任务并存储结果
    results[currentIndex] = await task();

    // 继续执行队列中的下一个
    await runNext();
  }

  // 启动 limit 个并发任务
  const workers = Array.from(
    { length: Math.min(limit, tasks.length) },
    () => runNext()
  );

  // 等待所有任务完成
  await Promise.all(workers);

  return results;
}

// 测试
async function delay(ms, value) {
  return new Promise(resolve =>
    setTimeout(() => resolve(value), ms)
  );
}

async function test() {
  const tasks = [
    () => delay(1000, 'A'),
    () => delay(1000, 'B'),
    () => delay(1000, 'C'),
    () => delay(1000, 'D'),
  ];

  console.time('runLimit');
  const results = await runLimit(tasks, 2);
  console.timeEnd('runLimit');  // 约 2000ms
  console.log(results);  // ['A', 'B', 'C', 'D']
}

test();
```

**方案二：队列 + 显式控制**

```javascript
function runLimit(tasks, limit) {
  return new Promise((resolve, reject) => {
    const results = [];
    let running = 0;  // 当前运行数
    let index = 0;    // 下一个任务索引
    let completed = 0;

    function run() {
      // 达到并发限制，等待
      if (running >= limit) return;

      // 所有任务已分配
      if (index >= tasks.length) {
        if (running === 0) resolve(results);
        return;
      }

      // 启动新任务
      running++;
      const currentIndex = index++;

      tasks[currentIndex]()
        .then(result => {
          results[currentIndex] = result;
          running--;
          completed++;

          // 检查是否全部完成
          if (completed === tasks.length) {
            resolve(results);
          } else {
            // 继续执行下一个
            run();
          }
        })
        .catch(reject);

      // 继续尝试启动任务（达到并发上限前）
      run();
    }

    // 启动初始任务
    run();
  });
}
```

**方案三：使用 async-pool 库（生产环境推荐）**

```javascript
// npm install async-pool
import asyncPool from 'async-pool';

const results = await asyncPool(
  2,           // 并发限制
  tasks,       // 任务列表
  task => task()
);
```
