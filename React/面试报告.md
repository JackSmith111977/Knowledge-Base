#### 1. 虚拟 DOM 完整结构

**问题**：React.createElement() 返回的对象结构是怎样的？

**完整答案**：

```javascript
{
  $$typeof: Symbol('react.element'),  // 标识这是 React 元素（安全验证）
  type: 'div',                         // 元素类型（字符串或组件函数）
  key: null,                           // 用于列表优化的 key
  ref: null,                           // 访问 DOM 或组件实例的引用
  props: {                             // 所有属性和子元素
    id: 'title',
    className: 'header',
    children: 'Hello'
  },
  _owner: null,                        // 指向创建该元素的组件（开发模式）
  _store: {}                           // 内部存储（开发模式）
}
```

**各属性详解：**

| 属性 | 说明 | 作用 |
|------|------|------|
| `$$typeof` | Symbol('react.element') | 安全验证，防止 XSS 攻击（恶意对象伪装成 React 元素） |
| `type` | 字符串或函数 | 元素类型：'div' 等宿主组件，或函数组件/类组件 |
| `key` | string/null | 列表 Diff 优化的唯一标识 |
| `ref` | object/function/null | 访问底层 DOM 节点或组件实例 |
| `props` | object | 所有属性 + children |
| `_owner` | Fiber 节点 | 开发模式下用于更准确的警告信息 |

**$$typeof 的安全作用：**

```javascript
// 防止 XSS 攻击
const maliciousElement = {
  type: 'div',
  props: { dangerouslySetInnerHTML: { __html: '<script>alert(1)</script>' } }
};

// React 会验证 $$typeof
if (element.$$typeof !== Symbol('react.element')) {
  throw new Error('Invalid React element');
}

// 由于 Symbol 是唯一的，恶意对象无法伪造
```

#### 2. Hooks 全局指针机制

**问题**：React 具体是如何「记住」每个 useState 的状态的？

**完整答案**：

**Hooks 链表结构：**

```javascript
// 组件内部维护一个链表
component.memoizedState = {
  memoizedState: 0,      // 当前状态值
  queue: null,           // 更新队列（存储 setState 的更新）
  next: null,            // 指向下一个 Hook（链表）
}

// 链表结构示意：
// Hook1 → Hook2 → Hook3 → Hook4
```

**简化版 useState 实现：**

```javascript
let currentHookIndex = 0;
let hooks = [];

function useState(initialState) {
  // 读取当前索引对应的 Hook
  const hook = hooks[currentHookIndex];

  // 如果是首次渲染，创建新 Hook
  if (!hook) {
    hooks.push({
      state: initialState,
      queue: []
    });
  }

  // 返回当前状态和更新函数（闭包）
  const setState = (newState) => {
    hooks[currentHookIndex].state = newState;
    // 触发重新渲染...
  };

  // 索引递增，指向下一个 Hook
  currentHookIndex++;

  return [hooks[currentHookIndex - 1].state, setState];
}

// 每次组件渲染前重置索引
function renderComponent() {
  currentHookIndex = 0;  // ← 关键：重置为 0
  // 组件代码调用 useState...
}
```

**React 内部实际实现：**

```javascript
// ReactCurrentOwner.currentHook 是全局指针
// 每次调用 useState 时：
// 1. 从 currentHook 读取当前节点
// 2. currentHook = currentHook.next（移动到下一个）

function updateStateReducer(reducer, initialArg, init) {
  // 获取当前 Hook 节点
  const hook = updateWorkInProgressHook();

  // hook.memoizedState 存储状态值
  // hook.queue 存储更新队列

  return hook.memoizedState;
}
```

#### 3. setState 引用稳定性

**问题**：setState 需要放入 useEffect 的依赖数组吗？

**完整答案**：

**不需要**。React 保证了 `setState` 函数是**持久化引用**，每次渲染都是同一个函数。

```jsx
// ✅ 正确：setState 不需要放入依赖数组
useEffect(() => {
  const timer = setInterval(() => {
    setCount(c => c + 1);  // 函数式更新，不依赖外部状态
  }, 1000);
  return () => clearInterval(timer);
}, []);  // 空依赖是安全的

// ⚠️ 虽然不会导致问题，但放入依赖数组是冗余的
useEffect(() => {
  // ...
}, [setState]);  // setState 永远不会变，所以不会触发额外执行
```

**React 源码保证：**

```javascript
// packages/react-reconciler/src/ReactFiberHooks.js
function mountState(initialState) {
  const hook = mountWorkInProgressHook();

  // 创建 dispatch 函数（一次性的）
  const dispatch = dispatchSetState.bind(null, fiber, queue);
  dispatch._reactName = 'dispatch';  // 标记为 React 内部函数

  hook.memoizedState = initialState;
  hook.queue = queue;

  return [hook.memoizedState, dispatch];  // dispatch 是持久化引用
}
```

**常见误区纠正：**

```jsx
// ❌ 错误理解：以为 setState 会变
function Counter() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    // 担心 setCount 变化导致重复执行
  }, [setCount]);  // 其实 setCount 永远不会变

  // ✅ 正确：不需要放入依赖数组
  useEffect(() => {
    const timer = setInterval(() => {
      setCount(c => c + 1);
    }, 1000);
    return () => clearInterval(timer);
  }, []);
}
```

#### 4. React 18 并发特性

**问题**：useTransition 和 useDeferredValue 是什么？如何使用？

**完整答案**：

**useTransition：标记非紧急更新**

```jsx
import { useState, useTransition } from 'react';

function SearchPage() {
  const [query, setQuery] = useState('');
  const [isPending, startTransition] = useTransition();

  // 紧急更新：立即执行（输入框）
  const handleChange = (e) => {
    setQuery(e.target.value);  // 立即更新
  };

  // 非紧急更新：可被中断（搜索结果）
  const handleQueryChange = (newQuery) => {
    startTransition(() => {
      setQuery(newQuery);  // 可被中断，让位于紧急更新
    });
  };

  return (
    <div>
      <input value={query} onChange={handleChange} />
      {isPending && <span>加载中...</span>}
      <SearchResults query={query} />
    </div>
  );
}
```

**useDeferredValue：延迟值更新**

```jsx
import { useState, useDeferredValue } from 'react';

function SearchPage() {
  const [query, setQuery] = useState('');
  const deferredQuery = useDeferredValue(query);  // 延迟版本

  return (
    <div>
      <input value={query} onChange={(e) => setQuery(e.target.value)} />
      {/* 使用延迟值，不会阻塞输入 */}
      <SearchResults query={deferredQuery} />
    </div>
  );
}
```

**两者区别：**

| 特性 | useTransition | useDeferredValue |
|------|---------------|------------------|
| 使用方式 | 包裹 setState | 包装值本身 |
| 返回值 | [isPending, startTransition] | deferredValue |
| 适用场景 | 需要控制更新时机 | 已有值需要延迟 |
| 底层原理 | 标记更新优先级 | 创建延迟副本 |

**实际应用：优化高频输入列表**

```jsx
function ExpensiveList({ items }) {
  const [filter, setFilter] = useState('');
  const deferredFilter = useDeferredValue(filter);

  // 过滤操作可能很耗时
  const filteredItems = items.filter(item =>
    item.name.includes(deferredFilter)
  );

  return (
    <div>
      <input
        value={filter}
        onChange={(e) => setFilter(e.target.value)}
        placeholder="搜索..."
      />
      {deferredFilter !== filter && <span>更新中...</span>}
      <ul>
        {filteredItems.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </div>
  );
}
```

#### 5. 请求取消机制（AbortController）

**问题**：如何在组件卸载或输入变化时取消 pending 请求？

**完整答案**：

**AbortController 基本用法：**

```javascript
const controller = new AbortController();
const { signal } = controller;

// 发起请求
fetch('/api/data', { signal })
  .then(response => response.json())
  .catch(error => {
    if (error.name === 'AbortError') {
      console.log('请求已取消');
    } else {
      console.error('请求失败:', error);
    }
  });

// 取消请求
controller.abort();
```

**React 中完整实现：**

```jsx
function SearchSuggestions({ query }) {
  const [suggestions, setSuggestions] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  useEffect(() => {
    // 创建 AbortController
    const controller = new AbortController();

    // 竞态问题处理：每次 effect 运行都创建新的 controller
    let isCurrentRequest = true;

    const fetchSuggestions = async () => {
      if (!query) {
        setSuggestions([]);
        return;
      }

      setLoading(true);
      setError(null);

      try {
        const response = await fetch(`/api/suggestions?q=${query}`, {
          signal: controller.signal
        });

        if (!response.ok) throw new Error('请求失败');

        const data = await response.json();

        // 只有当前请求是最新的才更新状态
        if (isCurrentRequest) {
          setSuggestions(data);
        }
      } catch (err) {
        if (err.name !== 'AbortError') {
          setError(err.message);
        }
      } finally {
        if (isCurrentRequest) {
          setLoading(false);
        }
      }
    };

    fetchSuggestions();

    // 清理函数：组件卸载或 query 变化时取消请求
    return () => {
      isCurrentRequest = false;
      controller.abort();
    };
  }, [query]);

  return (
    <div>
      {loading && <span>加载中...</span>}
      {error && <span className="error">{error}</span>}
      <ul>
        {suggestions.map(item => (
          <li key={item.id}>{item.text}</li>
        ))}
      </ul>
    </div>
  );
}
```

**自定义 Hook 封装：**

```jsx
function useFetchWithCancel(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const controller = new AbortController();
    let isMounted = true;

    const fetchData = async () => {
      try {
        const response = await fetch(url, {
          signal: controller.signal
        });

        if (!response.ok) throw new Error('请求失败');
        const result = await response.json();

        if (isMounted) {
          setData(result);
          setLoading(false);
        }
      } catch (err) {
        if (err.name !== 'AbortError' && isMounted) {
          setError(err.message);
          setLoading(false);
        }
      }
    };

    fetchData();

    return () => {
      isMounted = false;
      controller.abort();
    };
  }, [url]);

  return { data, loading, error };
}
```
